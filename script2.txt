local player = game.Players.LocalPlayer
local character = player.Character or player.CharacterAdded:Wait()

-- Cria um escudo esférico
local shield = Instance.new("Part")
shield.Shape = Enum.PartType.Ball -- Define a forma como esfera
shield.Size = Vector3.new(10, 10, 10) -- Ajuste o tamanho conforme necessário para envolver o personagem
shield.CanCollide = false -- Desativa a colisão geral para evitar problemas de movimento
shield.Transparency = 0.5 -- Define a transparência para visualização
shield.Anchored = false -- Permite que o escudo se mova com o jogador
shield.Massless = true -- Define o escudo como sem massa para não afetar o movimento
shield.Parent = character

-- Usa um `Weld` para conectar o escudo ao personagem
local weld = Instance.new("WeldConstraint")
weld.Part0 = shield
weld.Part1 = character.PrimaryPart -- Liga o escudo ao personagem
weld.Parent = shield

-- Define o comportamento de repulsão ao ser tocado
shield.Touched:Connect(function(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid and hit.Parent ~= character then
        local direction = (hit.Position - shield.Position).Unit
        humanoid:ApplyImpulse(direction * -1000) -- Ajuste a força de empurrão
    end
end)
